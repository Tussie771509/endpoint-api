const express = require('express'); const app = express(); const port = 3000; // Define a basic route app.get('/', (req, res) => { res.send('Hello, World!'); }); // Basic error handling middleware app.use((err, req, res, next) => { console.error(err.stack); res.status(500).send('Something went wrong!'); }); // Start the server app.listen(port, () => { console.log(`Server running at http://localhost:${port}/`); });
